## 定位问题

1.通过报错提示发现是编译`immortalwrt-mt798x/tools/mtools`时出现的报错  
2.通过这个[邮件](https://mail.gnu.org/archive/html/info-mtools/2022-04/msg00001.html)确定应该是在 4.0.39 这个版本修复了

## 下载合适的版本

通过第4行确定要下载的文件后缀为`.tar.bz2`  
本着用新不用旧的原则这里下载最新版本`4.0.44`

```Makefile
PKG_NAME:=mtools
PKG_VERSION:=4.0.35

PKG_SOURCE:=$(PKG_NAME)-$(PKG_VERSION).tar.bz2
PKG_SOURCE_URL:=@GNU/$(PKG_NAME)
PKG_HASH:=34769e173751d2f0d891a08c76c80427e929b8ee43438019b8666cc3d7a44749
```

## 更新 mtools 版本

1.在 `https://www.gnu.org/software/mtools` 下载合适的 mtools 版本  
2.使用 `sha256sum mtools-4.0.44.tar.bz2` 计算下载文件 sha256 的值

```diff
# 
# Copyright (C) 2012-2013 OpenWrt.org
#
# This is free software, licensed under the GNU General Public License v2.
# See /LICENSE for more information.
#
include $(TOPDIR)/rules.mk

PKG_NAME:=mtools
-PKG_VERSION:=4.0.35
+PKG_VERSION:=4.0.44

PKG_SOURCE:=$(PKG_NAME)-$(PKG_VERSION).tar.bz2
PKG_SOURCE_URL:=@GNU/$(PKG_NAME)
-PKG_HASH:=34769e173751d2f0d891a08c76c80427e929b8ee43438019b8666cc3d7a44749
+PKG_HASH:=37dc4df022533c3d4b2ec1c78973c27c7e8b585374c2d46ab64c6a3db31eddb8

HOST_BUILD_PARALLEL:=1

include $(INCLUDE_DIR)/host-build.mk

HOST_CONFIGURE_ARGS += --without-x

HOST_CONFIGURE_VARS += \
	ac_cv_header_iconv_h=no

define Host/Compile
	$(MAKE) -C $(HOST_BUILD_DIR) mcopy mmd
endef

define Host/Install
	$(INSTALL_BIN) $(HOST_BUILD_DIR)/mcopy $(STAGING_DIR_HOST)/bin/
	$(INSTALL_BIN) $(HOST_BUILD_DIR)/mmd $(STAGING_DIR_HOST)/bin/
endef

define Host/Clean
	rm -f $(STAGING_DIR_HOST)/bin/mcopy
	rm -f $(STAGING_DIR_HOST)/bin/mmd
endef

$(eval $(call HostBuild))
```
